    bool ListInsert(SqList* L, int i, ElemType e) {
    if (i > L->length + 1 || i < 1) {
        return false;
    }
    if (L->length >= MAXSIZE) {
        return false;
    }
    for (int j = L->length; j >= i; j--) {
        L->data[j] = L->data[j - 1];
    }
    L->data[i - 1] = e;
    L->length++;
    return true;
}

            pb = pb->next;
            free(u);

            free(u);
