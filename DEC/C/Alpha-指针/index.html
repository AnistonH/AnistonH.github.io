//将n个整数，循环后移m个位置
#define N 8
void cyc(int *p,int n,int m) {
    int i,k,temp;
    for(k=0; k<m; k++) {    //循环移位m次
        temp=*(p+n-1);  //暂存最后一个元素，空出那个位置
        for(i=0; i<n-1; i++)
            *(p+n-i-1)=*(p+n-i-2);//依次做：空出的位置保存前一个元素
        *p=temp;        //最前面的位置保存暂存的元素----循环移位一次
    }
}
int main() {

        scanf("%d",p+i);

    scanf("%d",&m);
    cyc(p,n,m);             //循环后移
    printf("移动后的数据为：\n");
    for(i=0; i<n; i++)
        printf("%-4d",*(p+i));
    printf("\n");
    return 0;
}
